const FeaturedCard = ({ images, text, index }) => {
  // Determine if this is an even or odd card (0-based index)
  const isEven = index % 2 === 0; // index 0, 2 → odd-positioned card (1st, 3rd) → row-reverse

  return (
    <div className="featured__card">
      <FeaturedImage
        featuredImages={images}
        animation={isEven ? "fade-left" : "fade-right"}
      />
      <FeaturedText
        featuredText={[text]}
        animation={isEven ? "fade-right" : "fade-left"}
      />
    </div>
  );
};

const FeaturedImage = ({ featuredImages, animation }) => {
  return (
    <div
      className="featured__image"
      data-aos={animation}
      data-aos-duration="1000"
      data-aos-easing="ease-out-quart"
    >
      {featuredImages.map((item) => (
        <img
          key={item.id}
          src={`/images/propertyimages/${item.img}`}
          alt={item.alt}
          className="featured__img"
        />
      ))}
    </div>
  );
};

const FeaturedText = ({ featuredText, animation }) => {
  return (
    <div
      className="featured__text"
      data-aos={animation}
      data-aos-duration="1000"
      data-aos-easing="ease-out-quart"
    >
      {featuredText.map((item, index) => (
        <React.Fragment key={index}>
          <h2 className="featured__text__title">{item.text_h2}</h2>
          <p className="featured__text__description">{item.text_p1}</p>
          <p className="featured__text__description">{item.text_p2}</p>
        </React.Fragment>
      ))}
    </div>
  );
};
